20  package org.jclouds.virtualbox.domain;
21  
22  import java.util.HashMap;
23  import java.util.Map;
24  
25  import org.virtualbox_4_1.VBoxException;
26  import org.virtualbox_4_1.jaxws.RuntimeFaultMsg;
27  
28  /**
29   * Maps the error codes in the Virtual Box Java API into enum values.
30   * <p/>
31   * <p/>
32   * To get the error code from a VBoxException, use:
33   * 
34   * <pre>
35   * try {
36   *    ...
37   *    ...
38   * }
39   * catch (VBoxException vboxException) {
40   *    RuntimeFaultMsg fault = (RuntimeFaultMsg) vboxException.getWrapped();
41   *    int faultCode = fault.getFaultInfo().getResultCode();
42   *    ErrorCode errorCode = ErrorCode.valueOf(faultCode);
43   * }
44   * </pre>
45   * 
46   * @author Mattias Holmqvist
47   */
48  public enum ErrorCode {
49  
50     VBOX_E_OBJECT_NOT_FOUND(2159738881L), VBOX_E_INVALID_VM_STATE(2159738882L), VBOX_E_VM_ERROR(2159738883L), VBOX_E_FILE_ERROR(
51           2159738884L), VBOX_E_IPRT_ERROR(2159738885L), VBOX_E_PDM_ERROR(2159738886L), VBOX_E_INVALID_OBJECT_STATE(
52           2159738887L), VBOX_E_HOST_ERROR(2159738888L), VBOX_E_NOT_SUPPORTED(2159738889L), VBOX_E_XML_ERROR(2159738890L), VBOX_E_INVALID_SESSION_STATE(
53           2159738891L), VBOX_E_OBJECT_IN_USE(2159738892L), VBOX_E_ACCESSDENIED(2147942405L), VBOX_E_POINTER(2147500035L), VBOX_E_FAIL(
54           2147500037L), VBOX_E_NOTIMPL(2147500033L), VBOX_E_OUTOFMEMORY(2147942414L), VBOX_E_INVALIDARG(2147942487L), VBOX_E_UNEXPECTED(
55           2147549183L), VBOX_E_UNKNOWN_ERROR_CODE(-1L), VBOX_E_ERROR_CODE_UNAVAILABLE(-2L);
56  
57     private long code;
58  
59     ErrorCode(long code) {
60        this.code = code;
61     }
62  
63     private static Map<Long, ErrorCode> table = new HashMap<Long, ErrorCode>();
64  
65     static {
66        for (ErrorCode errorCode : ErrorCode.values()) {
67           table.put(errorCode.code, errorCode);
68        }
69     }
70  
71     /**
72      * Returns an ErrorCode from the fault code given by the VirtualBox API.
73      * 
74      * @param vboxException
75      *           the exception to get the error code from.
76      * @return an ErrorCode representing the given fault code.
77      */
78     public static ErrorCode valueOf(VBoxException vboxException) {
79        final Throwable backend = vboxException.getWrapped();
80        if (backend instanceof RuntimeFaultMsg) {
81           final RuntimeFaultMsg faultCode = (RuntimeFaultMsg) backend;
82           final int resultCode = faultCode.getFaultInfo().getResultCode();
83           final ErrorCode errorCode = table.get(unsignedIntToLong(resultCode));
84           if (errorCode != null) {
85              return errorCode;
86           }
87           return VBOX_E_UNKNOWN_ERROR_CODE;
88        }
89        return VBOX_E_ERROR_CODE_UNAVAILABLE;
90     }
91  
92     private static long unsignedIntToLong(int faultCode) {
93        return faultCode & 0xffffffffL;
94     }
